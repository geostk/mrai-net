82 void TestSequence2D(int obj, int objsize, int imasize,
83  double tacq, double TE, double TR, double teta,
84  double b0, double db0, double b0def, int flagres, int flagdech, int filter,
85  char * fileout, char * objname, char * seqname, int display, int inhomogeneousB1)
86 {
87  char fic[80];
88  int i, j, k;
89  int ntx, nty, ntz; /* K space size */
90  double fovx, fovy, fovz; /* Fov size in meter */
91  OBJECT3D *object;
92  EXPERIENCE3D *expr;
93  SEQPARAM seqparam;
94  PPPVOLUME_COMPLEX_DOUBLE volrf;
95  PPPVOLUME_FLOAT volrecmod;
96 #ifdef HAVE_MPI
97  int rank;
98 #endif
99  fic[0] = '\0';
100 
101  printf("TestSequence2D start\n");
102  printf("B0 = %f\n", b0);
103 
104  if (objsize < imasize)
105  {
106  printf("The object has more pixel than the image !! \n");
107  exit(0);
108  }
109 
110  /* Object */
111  object = GetTestObject2D(obj, objsize, objsize, b0, db0, objname);
112  // AddNoiseToObject(object,5.0);
113  // ApplyMeanToObject(object,3);
114 
115  /* Experience */
116  expr = AllocExperience();
117  ntx = imasize; nty = ntx; ntz = 1;
118  fovx = 0.20; fovy = 0.20; fovz = 0.0020;
119  SetFovExperience(expr, fovx, fovy, fovz, 0.0, 0.0, 0.0);
120  SetAcqExperience(expr, ntx, nty, ntz, tacq*1e-3);
121  SetResonanceExperience(expr, flagres);
122  SetFlagdechExperience(expr, flagdech);
123  SetB0Experience(expr, b0);
124  SetB0DefExperience(expr, b0def);
125  expr->inhomogeneousB1 = inhomogeneousB1;
126 
127  /* Sequence */
128  SetSeqParamTE(&seqparam, TE);
129  SetSeqParamTR(&seqparam, TR);
130  SetSeqParamTeta(&seqparam, teta); /* pour test GE2D */
131 
132  /* Acquisition */
133  volrf = RunSequence(seqname, &seqparam, object, expr);
134  // WriteVolRF(volrf);
135  FreeObject(object);
136  // AddGaussianNoiseToRFVolume(volrf,0.001);
137  if (filter == 1) VolRFFiltering(volrf, HANNING);
138 
139  /* Reconstruction */
140  volrecmod = RecVolIFFTModuleFromVol(volrf);
141  volrf = NULL;/* Should be an IdVolFree function call, but volrf pointer if already free within IdVolIFFT function ! */
142 
143  /* Save & Display */
144 #ifdef HAVE_MPI
145  MPI_Comm_rank(MPI_COMM_WORLD,&rank);
146  if (rank==0) WriteVolRecUchar(volrecmod,fileout);
147 #else
148  if (display == 1) DisplayVolXY((PPPVOLUME)volrecmod, 0, "Recvol2D");
149  WriteVolRecUchar(volrecmod, fileout);
150 #endif
151  printf("min=%e max=%e \n", IdVolMinima(volrecmod), IdVolMaxima(volrecmod));
152  IdVolFree(volrecmod);
153 
154  printf("TestSequence2D End\n");
155 }
