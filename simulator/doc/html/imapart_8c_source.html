109 PPIMAGE IdImaAllocSubImage ( PPIMAGE im, int ox, int oy, int lx, int ly )
110 {
111  int x;
112  int y, j;
113  PPIMAGE_UCHAR ims = (PPIMAGE_UCHAR)im;
114  PPIMAGE_UCHAR imd = NULL;
115  PRIVATE_IMAGE * si;
116 
117  if ( (ox < 0) || (ox+lx>IdImaDimX(im)) || (lx<=0) ) {
118  IdErrno = IDERR_X_OUT_OF_IMAGE;
119  return 0;
120  }
121  if ( (oy < 0) || (oy+ly>IdImaDimY(im)) || (ly<=0) ) {
122  IdErrno = IDERR_Y_OUT_OF_IMAGE;
123  return 0;
124  }
125  if ( im==0 ) {
126  IdErrno = IDERR_POINTER_IS_NULL;
127  return 0;
128  }
129 
130  if ( IdLibidoType(im) != IMA) {
131  IdErrno = IDERR_WRONG_TYPE;
132  return(PPIMAGE)0;
133  }
134 
135  si=(PRIVATE_IMAGE *)calloc(1, sizeof(PRIVATE_IMAGE)+ly*sizeof(void *));
136  if(si){
137 
138  x = ox*IdSizeOfType(im);
139  si->DimX=si->UsedNbX=lx;
140  si->DimY=si->UsedNbY=ly;
141  si->Type=IdImaType(im);
142  si->subObject=1;
143  imd = (PPIMAGE_UCHAR)&(si[1]);
144  for ( y=oy, j=0; y<oy+ly; y++, j++ )
145  imd[j]= &(ims[y][x]);
146  IdSetFileName(imd,IdGetFileName(ims));
147  }
148  else IdErrno = IDERR_POINTER_IS_NULL; /* a mettre le bon code */
149 
150  /* Calcul du resultat *****/
151  return (PPIMAGE) imd;
152 }
