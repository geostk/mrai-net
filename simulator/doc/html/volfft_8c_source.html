49 static PPPVOLUME_COMPLEX
50 recon_3d(PPPVOLUME_COMPLEX volr)
51 {
52  int i, j, k, dimx, dimy, dimz;
53  PPPVOLUME_COMPLEX volx;
54 
55  dimx = IdVolDimX(volr) / 2;
56  dimy = IdVolDimY(volr) / 2;
57  dimz = IdVolDimZ(volr) / 2;
58 
59  volx =
60  (PPPVOLUME_COMPLEX) IdVolAlloc(IdVolDimX(volr),
61  IdVolDimY(volr),
62  IdVolDimZ(volr), VOL_COMPLEX);
63  if (volx == NULL)
64  IdErrPrintf("\n PROBLEME D'ALLOCATION ");
65 
66  for (i = 0; i < dimz; i++)
67  for (j = 0; j < dimy; j++)
68  for (k = 0; k < dimx; k++)
69 
70  {
71  volx[i][j][k] =
72  volr[i + dimz][j + dimy][k + dimx];
73  volx[i][j + dimy][k] =
74  volr[i + dimz][j][k + dimx];
75  volx[i + dimz][j + dimy][k] =
76  volr[i][j][k + dimx];
77  volx[i + dimz][j][k] =
78  volr[i][j + dimy][k + dimx];
79 
80  volx[i][j][k + dimx] =
81  volr[i + dimz][j + dimy][k];
82  volx[i][j + dimy][k + dimx] =
83  volr[i + dimz][j][k];
84  volx[i + dimz][j + dimy][k + dimx] =
85  volr[i][j][k];
86  volx[i + dimz][j][k + dimx] =
87  volr[i][j + dimy][k];
88  }
89  IdVolFree(volr);
90  return (volx);
91 }

93 static void
94 FFT1Dbis(PPPVOLUME_COMPLEX volc)
95 {
96  int i, j, k, Tx;
97  PSIGNAL_COMPLEX lign;
98  double sqtx;
99 
100  Tx = IdVolDimX(volc);
101  sqtx = sqrt((double) (Tx));
102  lign = (PSIGNAL_COMPLEX) IdSigAlloc(Tx, SIG_COMPLEX);
103 
104  for (i = 0; i < IdVolDimZ(volc); i++)
105  for (j = 0; j < IdVolDimY(volc); j++) {
106  /*
107  * remplissage de la ligne
108  */
109  for (k = 0; k < Tx; k++) {
110  lign[k] = volc[i][j][k];
111  }
112  /*
113  * calcul de la FFT de la ligne
114  */
115  IdSigFFT(lign, 1);
116  /*
117  * copie dans la ligne correspondante de volc
118  */
119  for (k = 0; k < Tx; k++) {
120  volc[i][j][k].re = lign[k].re / (sqtx);
121  volc[i][j][k].im = lign[k].im / (sqtx);
122  }
123  }
124  IdSigFree(lign);
125 }
