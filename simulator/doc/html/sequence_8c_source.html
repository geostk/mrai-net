96 PPPVOLUME_COMPLEX_DOUBLE
97  RunSequence(char * seqname, SEQPARAM *seqparam, OBJECT3D * objet, EXPERIENCE3D* expr)
98 {
99  int i,j,k;
100  PPPVOLUME_COMPLEX_DOUBLE volrf,volrfacu;
101  PSEQ pseq;
102 
103 /*******************************************************************************************************/
104 // Alexis Amadon, December 2005
105 // Introduce B1(r) inhomogeneities
106 
107 if (expr->inhomogeneousB1)
108  SetB1(expr);
109 
110 /*******************************************************************************************************/
111 
112 
113  pseq = GetSequenceByName(seqname);
114 
115 
116  volrfacu = (PPPVOLUME_COMPLEX_DOUBLE) IdVolAlloc(expr->ntx,expr->nty,expr->ntz,VOL_COMPLEX_DOUBLE);
117  if (!volrfacu)
118  {
119  FPRINTF(stderr,"Unable to allocate volrfacu!!\n");
120  exit(0);
121  }
122  for(i=0;i<expr->ntx;i++) for (j=0;j<expr->nty;j++) for (k=0;k<expr->ntz;k++)
123  {
124  volrfacu[k][j][i].re = 0.0;
125  volrfacu[k][j][i].im = 0.0;
126  }
127 
128  for (i=0;i<objet->nbcomponent;i++)
129  {
130  SetObjectActiveComponent(objet,i);
131  SetObjectExperienceComponent(expr,objet,i,expr->FLAG_DECH);
132  volrf = pseq(seqparam,expr);
133  AddComplexVolRF(expr,volrfacu,volrf);
134  IdVolFree(volrf);
135  }
136 
137  return(volrfacu);
138 }
