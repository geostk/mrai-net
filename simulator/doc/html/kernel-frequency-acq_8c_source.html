52 void KernelFrequency1(EXPERIENCE3D * expr, EVENT * event)
53 {
54  int ix, iy, iz;
55  float sgn[3];
56  double v;
57  int xmin;
58  int xmax;
59 
60 #ifdef HAVE_MPI
61  xmin= expr->mpi_slice_x0;
62  xmax= xmin+mpi_get_slice_width(expr, xmin);
63 #else
64  xmin = 0;
65  xmax = expr->nfovx;
66 #endif
67 
68  sgn[0] = 0.;
69  sgn[1] = 0.;
70  sgn[2] = 0.;
71 
72  if (event->op_flag == PLUS)
73  {
74  for (ix=xmin;ix<xmax;ix++)
75  for(iy=expr->nfovy -1;iy>=0;iy--)
76  for(iz=expr->nfovz -1;iz>=0;iz--)
77  { /* Spins summation */
78  /* T2* in the transverse plane xy */
79  v = exp(-fabs(expr->to*G_Hz_T*expr->deltaB0[ix][iy][iz]));
80  sgn[0] += (float) expr->magstate[0][ix][iy][iz]* (float)v;
81  sgn[1] += (float) expr->magstate[1][ix][iy][iz]* (float)v;
82  sgn[2] += (float) expr->magstate[2][ix][iy][iz];
83  }
84  }
85  else if (event->op_flag == MINUS)
86  {
87  for (ix=xmin;ix<xmax;ix++)
88  for(iy=expr->nfovy -1;iy>=0;iy--)
89  for(iz=expr->nfovz -1;iz>=0;iz--)
90  { /* Spins summation */
91  /* T2* in the transverse plane xy */
92  v = exp(-fabs(expr->to*G_Hz_T*expr->deltaB0[ix][iy][iz]));
93  sgn[0] -= (float) expr->magstate[0][ix][iy][iz]* (float) v;
94  sgn[1] -= (float) expr->magstate[1][ix][iy][iz]* (float) v;
95  sgn[2] -= (float) expr->magstate[2][ix][iy][iz];
96  }
97  }
98 
99 /* First k space point saving */
100 #ifndef OSCILLO
101  expr->sgn[0][0][event->nu_pointy][event->nu_pointz] += sgn[0];
102  expr->sgn[1][0][event->nu_pointy][event->nu_pointz] += sgn[1];
103  expr->sgn[2][0][event->nu_pointy][event->nu_pointz] += sgn[2];
104 #else
105  expr->sgn[0][0][event->nu_pointy][event->nu_pointz] = sgn[0];
106  expr->sgn[1][0][event->nu_pointy][event->nu_pointz] = sgn[1];
107  expr->sgn[2][0][event->nu_pointy][event->nu_pointz] = sgn[2];
108 #endif
109 }

119 void KernelFrequency2(EXPERIENCE3D * expr, EVENT * event)
120 {
121  double v;
122  double x, y, z, dw,edw;
123  double p6x, p6y, p6z, p9x, p9y, p9z;
124  double e1, e2, teta, rmnx, r1, r2;
125  double *e0xyz, *e1xyz, *e2xyz;
126  int ix, iy, iz, it;
127  float *sgn[3];
128  int xmin;
129  int xmax;
130 
131 #ifdef HAVE_MPI
132  xmin=expr->mpi_slice_x0;
133  xmax=xmin+mpi_get_slice_width(expr, xmin);
134 #else
135  xmin = 0;
136  xmax = expr->nfovx;
137 #endif
138 
139  p6x = G * event->ax;
140  p9x = expr->xf + expr->obx - event->axoff;
141  p6y = G * event->ay;
142  p9y = expr->yf + expr->oby - event->ayoff;
143  p6z = G * event->az;
144  p9z = expr->zf + expr->obz - event->azoff;
145 
146  /* RF signal allocation */
147  sgn[0] = (float *)malloc (expr->ntx * sizeof(float));
148  sgn[1] = (float *)malloc (expr->ntx * sizeof(float));
149  sgn[2] = (float *)malloc (expr->ntx * sizeof(float));
150 
151  /* RF signal point acquisition */
152  for (it = 1; it < expr->ntx; it++)
153  {
154  sgn[0][it] = 0.;
155  sgn[1][it] = 0.;
156  sgn[2][it] = 0.;
157  /* T2* account */
158  expr->to = expr->to + expr->pt;
159 
160  for (ix = xmin; ix < xmax; ix++) {
161  x = expr->px * ix - p9x;
162  for (iy = expr->nfovy - 1; iy >= 0; iy--) {
163  y = expr->py * iy - p9y;
164  for (iz = expr->nfovz - 1; iz >= 0; iz--)
165  { /* Update of the spin states (Bloch) before the it point acquisition */
166  z = expr->pz * iz - p9z;
167 
168  e0xyz = &(expr->magstate[0][ix][iy][iz]);
169  e1xyz = &(expr->magstate[1][ix][iy][iz]);
170  e2xyz = &(expr->magstate[2][ix][iy][iz]);
171 
172  /* dw is given in rad/s */
173 // dw =(p6x * x)+(p6y * y)+(p6z * z)-(expr->G_edw)+G_rad_T*expr->G_deche[ix][iy][iz];
174  edw = expr->G_edw *(expr->B0+1.*(event->ax*x + event->ay*y + event->az*z) + expr->G_deche[ix][iy][iz]);
175  dw = G_rad_T*(event->ax*x + event->ay*y + event->az*z + expr->G_deche[ix][iy][iz]) - edw;
176  teta = dw * expr->pt;
177 
178  e1 = exp(-expr->pt / expr->t1[ix][iy][iz]);
179  e2 = exp(-expr->pt / expr->t2[ix][iy][iz]);
180 
181  r1 = e2 * cos(teta);
182  r2 = e2 * sin(teta);
183 
184  rmnx = r1 * e0xyz[0] + r2 * e1xyz[0];
185  e1xyz[0] = -r2 * e0xyz[0] + r1 * e1xyz[0];
186  e2xyz[0] = e1 * e2xyz[0] + expr->ro[ix][iy][iz] * (1 - e1);
187  e0xyz[0] = rmnx;
188 
189  v = exp(-fabs(expr->to*G_Hz_T*expr->deltaB0[ix][iy][iz]));
190  if (event->op_flag == PLUS)
191  { /* Spins summation */
192  /* T2* in the transverse plane xy */
193  sgn[0][it] += (float) e0xyz[0] * (float) v;
194  sgn[1][it] += (float) e1xyz[0] * (float) v;
195  sgn[2][it] += (float) e2xyz[0];
196  }
197  else if (event->op_flag == MINUS)
198  { /* Spins summation */
199  /* T2* in the transverse plane xy */
200  sgn[0][it] -= (float) e0xyz[0] * (float) v;
201  sgn[1][it] -= (float) e1xyz[0] * (float) v;
202  sgn[2][it] -= (float) e2xyz[0];
203  }
204 
205  }
206  }
207  }
208  }
209 
210 #ifndef OSCILLO
211  for (it = 1; it < expr->ntx; it++) {
212  expr->sgn[0][it][event->nu_pointy][event->nu_pointz] += sgn[0][it];
213  expr->sgn[1][it][event->nu_pointy][event->nu_pointz] += sgn[1][it];
214  expr->sgn[2][it][event->nu_pointy][event->nu_pointz] += sgn[2][it];
215  }
216 #else
217  for (it = 1; it < expr->ntx; it++) {
218  expr->sgn[0][it][event->nu_pointy][event->nu_pointz] = sgn[0][it];
219  expr->sgn[1][it][event->nu_pointy][event->nu_pointz] = sgn[1][it];
220  expr->sgn[2][it][event->nu_pointy][event->nu_pointz] = sgn[2][it];
221  }
222 #endif
223 
224  free(sgn[0]);
225  free(sgn[1]);
226  free(sgn[2]);
227 
228  if (event->spoiling_flag==1)
229  /* Numerical spoiling : Canceling the transverse magnetisation */
230  for (ix=xmin;ix<xmax;ix++)
231  for(iy=expr->nfovy -1;iy>=0;iy--)
232  for(iz=expr->nfovz -1;iz>=0;iz--)
233  {
234  expr->magstate[0][ix][iy][iz]=0.0;
235  expr->magstate[1][ix][iy][iz]=0.0;
236  }
237 
238 }
