145 * nom : fich_res

166 int IdImaLabel(PPIMAGE_UCHAR im, PPIMAGE_SHORT imo, int surf_min, FILE *fich_res, FILE *fich_mod, int connexite)
167 
168 
169 
170 
171  /* connexite = 1 : 8-connexite connexite = 0 :4- connexite */
172 {
173 unsigned char h;
174 short niv,couleur,ctr[1024];
175 int compt=0,col,col1;
176 int y,i,j,nbp,nbp2;
177  /* Controle des l'images **/
178  if ( !im || IdImaType(im) != IMA_UCHAR ){
179  IdErrno = IDERR_WRONG_TYPE;
180  return -1;
181  }
182  if ( !imo || IdImaType(imo) != IMA_SHORT ){
183  IdErrno = IDERR_WRONG_TYPE;
184  return -1;
185  }
186 
187 surfmini=surf_min;
188 surf_tot=0;surf_gros=0;nb_petit=0;nb_gros=0;couleur=0;
189 pt_cou=cou; pt_sup=sup;
190 for(y=0;y<512;y++) { *pt_cou++ = -1; *pt_sup++ = -1; }
191 pt_cou=cou; pt_sup=sup;
192 pre_mod=Malloc(modele); premier=Malloc(connexe); dernier_cx=Malloc(connexe);
193 cou_mod=pre_mod; courant=premier; premier->suivant=dernier_cx;
194 for(y=0;y<=IdImaDimY(im);y++)
195  {
196  DEBUG(IdPrintf("\n%d : ",y);)
197  courant=premier;
198  while(courant->suivant != dernier_cx)
199  {
200  h=0;
201  courant=courant->suivant;
202  while(courant->trans_cours[h] != -1)
203  {
204  courant->trans[h] = courant->trans_cours[h];
205  courant->trans[++h] = -1;
206  }
207  courant->trans_cours[0] = -1;
208  }
209  pt_cou=cou; pt_sup=sup;
210  while(*pt_cou != -1) *pt_sup++ = *pt_cou++;
211  *pt_sup = -1;
212  pt_cou=cou; pt_sup=sup;
213  if(y != IdImaDimY(im)) _label_trans(im,y);
214  else {*pt_cou++ = -1; *pt_cou = -1; pt_cou=cou; pt_sup=sup; }
215  while((*pt_cou != -1)&&(*pt_sup != -1))
216  {
217  if((*(pt_cou+1)+connexite) < *pt_sup) _label_debut(imo,y,&couleur);
218  else if(*pt_cou > (*(pt_sup+1)+connexite)) _label_fin(fich_res,fich_mod,y);
219  else _label_poursuite(imo,y,connexite);
220  }
221  if(*pt_cou != -1) while(*pt_cou != -1) _label_debut(imo,y,&couleur);
222  else while(*pt_sup != -1) _label_fin(fich_res,fich_mod,y);
223  }
224 _label_resultat(fich_res);
225 der_mod=cou_mod;cou_mod=pre_mod;
226 for(col=0;col<1024;col++) ctr[col]=0;
227 while(cou_mod != der_mod) {
228  cou_mod=pre_mod->suivant;
229  if(cou_mod->surface >= surf_min){
230  compt++;
231 /* IdPrintf("\n%3d : ",compt); */
232  col=0;
233  while(cou_mod->coul[col]) {
234  col1=cou_mod->coul[col++];
235 /* IdPrintf("%d ",col1); */
236  ctr[col1]=compt;
237  if(col >= NB_MAX_COUL-2)
238  {IdErrPrintf("NB_MAX_COUL trop petit\n");IdExit(0);}
239  }
240  }
241  pre_mod=cou_mod;
242  }
243 
244 nbp=0;
245 nbp2=0;
246 for(i=0;i<IdImaDimY(imo);i++)
247  for(j=0;j<IdImaDimX(imo);j++){
248  niv = imo[i][j];
249  if(niv){
250  imo[i][j] = ctr[niv];
251  nbp2++;
252  if(ctr[niv]) nbp++;
253  /* else IdPrintf("N %d\t",niv); */
254  }
255  }
256 
257 if((nbp != nbp2) && (surfmini < 2)) IdErrPrintf("Pb Tableau des niveaux\n");
258 IdPrintf("Surface totale %d, des 'gros' objets: %d, Pts image: %d\n",surf_tot,surf_gros,nbp);
259 /*
260 i=1; while(ctr[i]) {IdPrintf("%d --> %d\n",i,ctr[i]); i++;}
261 for(i=1;i<1024;i++) if(ctr[i]) {IdPrintf("%d --> %d\n",i,ctr[i]);}
262 */
263 
264 free((char *)premier);free((char *)dernier_cx);
265 der_mod=cou_mod;cou_mod=pre_mod;
266 while(cou_mod != der_mod)
267  { cou_mod=pre_mod->suivant; free((char *)pre_mod);pre_mod=cou_mod;}
268 free((char *)der_mod);
269 return(nb_gros);
270 }

273  /* Determination des transitions de la ligne traitee */
274 static void
275  _label_trans(PPIMAGE_UCHAR im, int lig)
276 {
277 int x;
278 
279 DEBUG( IdPrintf("Ligne %d\n",lig); )
280 if( im[lig][0] ) { *pt_cou=0; /*IdPrintf("1ere colonne non vide\n");*/}
281 for(x=0;x<IdImaDimX(im)-1;x++)
282  {
283  if(im[lig][x] < im[lig][x+1]) *pt_cou = x+1;
284  else if(im[lig][x] > im[lig][x+1]) { *(++pt_cou) = x; pt_cou++;}
285  }
286 if(im[lig][x]) { *(++pt_cou) = IdImaDimX(im)-1; pt_cou++; }
287 *pt_cou++ = -1; *pt_cou = -1;
288 pt_cou=cou; pt_sup=sup;
289 }

327  _label_fin(FILE *fich_res, FILE *fich_mod, int lig)
328 {

342  {
343  DEBUG(IdPrintf("! ");)

359  cou_mod->ydeb = fus_cx->ydeb; cou_mod->yfin = lig-1;
360 /* ----- Elimination du connexe ----- */
361  precsup->suivant = fus_cx->suivant;
362  free((char *)fus_cx);
363 /* ----- Affichage des parametres de chaque modele ----- */
364  if(cou_mod->surface >= surfmini) {
365  nb_gros++;surf_gros += cou_mod->surface;
366  fprintf(fich_mod,"%d\t%d\t",cou_mod->xdeb,cou_mod->ydeb);
367  fprintf(fich_mod,"%d\t %d\t%d\n",cou_mod->xfin,cou_mod->yfin,nb_gros);
368  fprintf(fich_res,"Objet No : %d\tSurface = %d\n",nb_gros,cou_mod->surface);
369  fprintf(fich_res,"\tXmin = %d\tYmin = %d",cou_mod->xdeb,cou_mod->ydeb);
370  fprintf(fich_res,"\tXmax = %d\tYmax = %d\n",cou_mod->xfin,cou_mod->yfin);
371  }
372  else nb_petit++;
373  }
374 pt_sup += 2;
375 }

492 /******************************************************************************/
493 /* --- Mise a jour des parametres du connexe en cours : FUSION --- */
494 /* --- Fusion de deux regions qui appartenaient a des connexes differents --- */
495 static void
496  _label_maj_fus(void)
497 {
498 unsigned char k,j;
499 
500 k=0;j=0;
501 while(fus_cx->trans_cours[k] != -1) k++;
502 while(courant->trans_cours[j] != -1)
503  { fus_cx->trans_cours[k] = courant->trans_cours[j]; j++; k++; }
504 fus_cx->trans_cours[k++] = *pt_cou;
505 fus_cx->trans_cours[k] = -1;
506 k=0;j=0;
507 while(fus_cx->trans[k] != -1) k++;
508 while(courant->trans[j] != -1)
509  {
510  DEBUG( IdPrintf("\nNo %d",courant->trans[j]);)
511  fus_cx->trans[k] = courant->trans[j]; j++; k++;
512  }
513 fus_cx->trans[k] = -1;
514 fus_cx->surface += courant->surface;
515 fus_cx->xdeb = min(min(fus_cx->xdeb,courant->xdeb),*pt_cou);
516 fus_cx->xfin = max(max(fus_cx->xfin,courant->xfin),*(pt_cou+1));
517 fus_cx->ydeb = min(fus_cx->ydeb,courant->ydeb);
518 precedent->suivant = courant->suivant;
519 free((char *)courant);
520 courant=fus_cx;
521 }
