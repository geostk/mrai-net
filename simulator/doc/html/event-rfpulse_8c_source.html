73 void DoPulseSinc(EXPERIENCE3D * expr,EVENT * event, double angle, double duration, int nb_points, int nb_lobs)
74 {
75 
76  double *fpul;//save the basic rectangular RF pulse used to define sinc shaped pulse
77  int ip;
78  double b1,fact,som,B,A,trp,b1v,angle_elementaire,agz;
79  FILE *fic;
80 
81  agz=event->agz;// saving the gradient value because event->agz set to zero in init_evente3D
82 
83  // Computation of the table containing the set of nb_points rectangular RF pulse defining the sinc one
84  // Of the rectangular pulse have the same duration (duration/nb_points) and their amplitude vary
85  // The angle sum is equal to the global angle
86  fpul=(double *)malloc(nb_points*sizeof(double));
87  som=0;
88  A = duration/2.0;
89  B=nb_lobs*M_PI/A;
90  for(ip=0;ip<nb_points;ip++)
91  {
92  trp=ip*duration/nb_points;
93  if(trp==A) fpul[ip]=1;
94  else fpul[ip]=sin(B*(trp-A))/(B*(trp-A));
95  som=som+fpul[ip];
96  }
97  fact=nb_points/som;
98 
99  // Temporary saving
100  fic=fopen("sinus.txt","w");
101  for(ip=0;ip<nb_points;ip++) fprintf(fic,"%f\n",fpul[ip]);
102  fclose(fic);
103 
104  // Applying the nb_points pulses
105  agz=event->agz;
106  b1=angle/(G*duration*1e-6)*fact;
107  for(ip=0;ip<nb_points;ip=ip++)
108  {
109  b1v=b1*fpul[ip];
110  angle_elementaire=b1v*G*duration*1e-6/nb_points;
111  event->agz=agz;
112  DoPulseRect(expr,event, angle_elementaire, duration/nb_points);
113  }
114  free(fpul);
115 }

146  fscanf(fic,"%d ",&nb_points);
147  fpul=(double *)malloc(nb_points*sizeof(double));
148  for(ip=0;ip<nb_points;ip++) fscanf(fic,"%lf ",&fpul[ip]);
149 
150  fclose(fic);
151  // Normalisation
152  som=0;
153  for(ip=0;ip<nb_points;ip++) som=som+fpul[ip];
154  fact=nb_points/som;
155 
156  // Apply the pulses
157  agz=event->agz;
158  som=0;
159  b1=angle/(G*duration*1e-6)*fact;
160  for(ip=0;ip<nb_points;ip=ip++)
161  {
162  b1v=b1*fpul[ip];
163  angle_elementaire=b1v*G*duration*1e-6/nb_points;
164  event->agz=agz;
165  printf("ipnum=%d : %d \n",ip,nb_points);
166  DoPulseRect(expr,event, angle_elementaire, duration/nb_points);
167  som=som+angle_elementaire;
168  }
169  free(fpul);
170 }
