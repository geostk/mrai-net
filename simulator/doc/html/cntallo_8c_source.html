125 PCONTOUR IdCntAlloc(int sx, int ty)
126  /* retourne PCONTOUR si OK, 0 si erreur */
127  /* ty = type de donnees: CNT_FLOAT ... */
128 {
129  PRIVATE_CONTOUR *si;
130  PCONTOUR pi;
131 
132  if(!(ty & IDO_TYPE)==CNT) {
133  IdErrPrintf("Allocation impossible. Objet de type incompatible.");
134  IdErrno=IDERR_WRONG_TYPE;
135  return((PCONTOUR)0);
136  }
137 
138  si=(PRIVATE_CONTOUR *)calloc(1, sizeof(PRIVATE_CONTOUR) + sx * IdTypeSize(ty));
139  if(si){
140  si->DimX=sx;
141  si->Type=ty;
142  si->_fichier=0; /* JPR */
143  si->_message=0; /* JPR */
144  pi=(PCONTOUR)&(si[1]);
145  }else{
146  IdErrPrintf("Echec d'allocation de contour. Memoire saturee.");
147  IdErrno=IDERR_ALLOC_CNT;
148  return((PCONTOUR)0); /* JPR */
149  }
150 
151  return(pi);
152 }
