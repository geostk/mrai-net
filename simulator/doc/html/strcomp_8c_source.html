55  * RETOUR: type : comp

58  * PARAMETRES:
59  * nom : a, b
60  * type : char *
61  * role : chaines a comparer.
62  *
63  * FICHIER: strcomp.c
64  *
65  * ******************************************************** END DESCRIPTION
66  */
67 
68 int
69 IdStrCmpNoCase(char *s1, char *s2)
70 {
71  do {
72  int c;
73  if ((c = UP(*s1) - UP(*s2)) != 0)
74  return c;
75  s1++;
76  s2++;
77  } while ((*s1) || (*s2));
78  return 0;
79 }
