122 void SeqInversionRecuperation1D_0(SEQPARAM *seqparam,EXPERIENCE3D*expr)
123 {
124  int ntx;
125  double fovx;
126  double te,tr,tacq,ti; /* in ms */
127  double gx,gy,gz;
128  double acq_delay,timp; /* in ms */
129  int npy, npz;
130  int i;
131  EVENT* event, event_r;
132  event = &event_r;
133 
134  printf("Starting SeqInversionRecuperation1D\n");
135  ntx = expr->ntx;
136  tacq = expr->readout_time*1.e+3;
137  fovx = expr->fovx;
138 
139  te = seqparam->te;
140  tr = seqparam->tr;
141  ti = seqparam->ti;
142 
143  CheckTimesIR(tacq,te,tr,ti);
144 
145  acq_delay = (te/2) - (tacq/2. + tacq/2./(ntx-1));
146  timp=0.3; /*RF pulse duration in ms */
147 
148  gx=(ntx-2.)/tacq/G_kHz_G/fovx;
149  gy=0.0;
150  gz=0.0;
151  npy=0;npz=0;
152 
153  SetSpoilingFlag(event,ACTIVE); /* XY Spoiling after each line acquisition */
154  DoWaiting(expr,event,ATT_INIT); /* To get an initial Magnetization M0 */
155  ResetToExperience(expr);
156 
157  for(i=0;i<10;i++) /* Repetition cycle to take TR into account */
158  {
159  DoPulseRect(expr,event, 180.,timp);
160  DoWaiting(expr,event,ti);
161  DoPulseRect(expr,event, 90., timp);
162  DoGradient(expr,event, te/2.,gx,gy,gz);
163  DoPulseRect(expr,event,180.,timp);
164  DoGradient(expr,event, acq_delay+tacq,gx,gy,gz);
165  DoWaiting(expr,event,tr-te-tacq/2.0);
166  }
167 
168  DoPulseRect(expr,event, 180.,timp);
169  DoWaiting(expr,event,ti);
170  DoPulseRect(expr,event, 90.,timp);
171  DoGradient(expr,event, te/2.,gx,gy,gz);
172  DoPulseRect(expr,event,180.,timp);
173  DoAcqFrequenceX( expr,event, gx, npy, npz,acq_delay, MINUS);
174 
175  printf("Ending SeqInversionRecuperation1D\n");
176 }

186 void SeqInversionRecuperation2D_0(SEQPARAM *seqparam,EXPERIENCE3D*expr)
187 {
188  int ntx,nty;
189  double fovx,fovy;
190  double te,tr,ti,tacq,timp,acq_delay,tg; /* ms */
191  double gx,gy,gz,gp;
192  int npy, npz;
193  EVENT* event, event_r;
194  event = &event_r;
195 
196  printf("Starting SeqInversionRecuperation2D \n");
197  ntx = expr->ntx;
198  nty = expr->nty;
199  tacq = expr->readout_time*1.e+3;
200  fovx = expr->fovx;
201  fovy = expr->fovy;
202 
203  ti = seqparam->ti;
204  te = seqparam->te;
205  tr = seqparam->tr;
206 
207  CheckTimesIR(tacq,te,tr,ti);
208  timp=0.3; /* Pulse duration in ms */
209  acq_delay = (te/2) - (tacq/2. + tacq/2./(ntx-1));
210  tg = te/2.;
211  gx = (ntx-2.)/tacq/G_kHz_G/fovx;
212  gy = 2*M_PI*(nty-1)/G_rad_G/fovy/(tg/1000);
213  gz = 0.0;
214  npz= 0;
215 
216  SetSpoilingFlag(event,ACTIVE); /* XY Spoiling after each line acquisition */
217 
218  DoWaiting(expr,event,ATT_INIT); /* To get an initial Magnetization M0 */
219  for (npy = 0; npy < expr->nty; npy++)
220  {
221  ResetToExperience(expr);
222  DoPulseRect(expr,event, 180.,timp);
223  DoWaiting(expr,event,ti);
224 
225  DoPulseRect(expr,event, 90.,timp);
226  gp = (gy) * (npy - (nty) / 2) / (nty);
227  DoGradient(expr,event,tg,gx,gp,gz);
228 
229  DoPulseRect(expr,event,180.,timp);
230 
231  DoAcqFrequenceX( expr,event, gx, npy, npz,acq_delay, MINUS);
232 
233  DoWaiting(expr,event,tr-te-tacq/2.0);
234  }
235  printf("Ending SeqInversionRecuperation2D\n");
236 }
