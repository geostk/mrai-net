45 void WriteVolRF(PPPVOLUME_COMPLEX_DOUBLE volrf)
46 {

48  int i,j,k;
49  int X,Y,Z;
50 
51  X=IdVolDimX(volrf);
52  Y=IdVolDimY(volrf);
53  Z=IdVolDimZ(volrf);
54 
55  re = (PPPVOLUME_DOUBLE) IdVolAlloc(X,Y,Z,VOL_DOUBLE);
56  im = (PPPVOLUME_DOUBLE) IdVolAlloc(X,Y,Z,VOL_DOUBLE);
57  for (k=0;k<Z;k++) for (i=0;i<Y;i++) for (j=0;j<X;j++)
58  {
59  re[k][i][j] = volrf[k][i][j].re;
60  im[k][i][j] = volrf[k][i][j].im;
61  }
62  IdVolWriteRawFile("rf_re.vol",(PPPVOLUME)re);
63  IdVolWriteRawFile("rf_im.vol",(PPPVOLUME)im);
64 
65  IdVolFree(re);
66  IdVolFree(im);

78 void WriteVolRecUchar(PPPVOLUME_FLOAT volrec, char * volname)
79 {
80  int X,Y,Z;
81  PPPVOLUME_UCHAR voluchar;
82  PPIMAGE_UCHAR imauchar;
83  double min=0.0;
84  double max=0.0;
85  double dyn;
86  int i,j,k;
87 
88  IdVolWriteRawFile("recfloat.raw",(PPPVOLUME)volrec);
89  X=IdVolDimX(volrec);
90  Y=IdVolDimY(volrec);
91  Z=IdVolDimZ(volrec);
92 
93  IdVolMinMax((PPPVOLUME)volrec,&min,&max);
94  dyn=(max-min);
95  if (Z==1) /* Writing a 2D image */
96  {
97  imauchar = (PPIMAGE_UCHAR) IdImaAlloc(X,Y,IMA_UCHAR);
98  for (i=0;i<Y;i++) for (j=0;j<X;j++)
99  imauchar[i][j]=(unsigned char)(((volrec[0][i][j]-min)/dyn)*255.0+0.5);
100  IdImaWriteRawFile(volname,(PPIMAGE)imauchar);
101  IdImaFree(imauchar);
102  }
103  else
104  {
105  voluchar = (PPPVOLUME_UCHAR) IdVolAlloc(X,Y,Z,VOL_UCHAR);
106  for (k=0;k<Z;k++) for (i=0;i<Y;i++) for (j=0;j<X;j++)
107  voluchar[k][i][j]=(unsigned char)(((volrec[k][i][j]-min)/dyn)*255.0+0.5);
108  IdVolWriteRawFile(volname,(PPPVOLUME)voluchar);
109  IdVolFree(voluchar);
110  }
111 }

171 void MagstateWatch(EXPERIENCE3D* expr,char * filename)
172 {
173 int g,ii;
174 FILE *fic;
175 int x,y;
176 
177 x=expr->nfovx/2;
178 y=expr->nfovy/2;

181  {
182  for(ii=0;ii<=2;ii++)
183  {
184  fprintf(fic,"%lf ",expr->magstate[ii][x][y][g]);
185  }
186  fprintf(fic,"%lf ",sqrt(expr->magstate[0][x][y][g]*expr->magstate[0][x][y][g]+expr->magstate[1][x][y][g]*expr->magstate[1][x][y][g]));
187  fprintf(fic,"%lf ",atan2(expr->magstate[1][x][y][g],expr->magstate[0][x][y][g])*180/M_PI);
188  fprintf(fic,"\n");
189  }
190 fclose(fic);
191 }
