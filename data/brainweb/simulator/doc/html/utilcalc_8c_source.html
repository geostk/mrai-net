155 IdCalcAngleFromPoints(double xB, double yB,
156  double xA, double yA,

158 {
159  double alpha, beta, gamma;
160  double xAn, yAn, xBn, yBn, xCn, yCn;
161 
162  /*
163  * changement de repere :
164  * translation de l'origine de l'ancien repere en A
165  */
166  xAn = 0.;
167  yAn = 0.;
168  xBn = xB - xA;
169  yBn = yB - yA;
170  xCn = xC - xA;
171  yCn = yC - yA;
172 
173  /*
174  * beta est l'angle forme entre l'axe X' du nouveau repere
175  * et le segment [AB]
176  */
177  beta = atan2(yBn, xBn);
178 
179  /*
180  * gamma est l'angle forme entre l'axe X' du nouveau repere
181  * et le segment [AC]
182  */
183  gamma = atan2(yCn, xCn);
184 
185  /*
186  * alpha = beta - gamma
187  */
188  alpha = beta - gamma;
189 
190  alpha = -alpha;
191  return (alpha);
192 }

301 IdCalcRacinesPoly2(double a, double b, double c)
302 {
303  double d, signe, den;
304  COMPLEX q;
305  PVECTOR_COMPLEX racines;
306 
307  racines = (PVECTOR_COMPLEX) IdVectAlloc(2, VECT_COMPLEX);
308  if (!racines) {
309  IdErrPrintf("\n Erreur allocation solution du 2nd degre");
310  IdExit(1);
311  }
312 
313  d = (double) (b * b - 4.0 * a * c);
314  if (b >= 0.0)
315  signe = 1.0;
316  else
317  signe = -1.0;
318 
319  if (d >= 0.0) {
320  q.re = -(b + signe * sqrt(d)) / 2;
321  q.im = 0.0;
322  } else {
323  q.re = -b / 2;
324  q.im = -signe * sqrt(-d) / 2;
325  }
326  den = q.im * q.im + q.re * q.re;
327  racines[0].re = q.re / a;
328  racines[0].im = q.im / a;
329  racines[1].re = c * q.re / den;
330  racines[1].im = -c * q.im / den;
331 
332  return (racines);
333 }

344 PVECTOR_COMPLEX
345 IdCalcRacinesPoly3(double a, double b, double c, double d)
346 {
347  PVECTOR_COMPLEX racines;
348  double q2, Q, R, A=0., B=0., C=0., S, T;
349  double teta, teta3, A3, Q3, KPI = (double) M_PI * 2 / 3, signe;
350 
351  if (!a) {
352  IdErrPrintf("\n Erreur degre du polynome");
353  IdExit(1);
354  } else {
355  A = b / a;
356  B = c / a;
357  C = d / a;
358  }
359 
360  racines = (PVECTOR_COMPLEX) IdSigAlloc(3, VECT_COMPLEX);
361 
362  if (!racines) {
363  IdErrPrintf("\n Erreur allocation solution 3eme degre");
364  IdExit(1);
365  }
366 
367  R = (A * A * A * 2 - A * B * 9 + C * 27) / 54;
368  Q = (A * A - B * 3) / 9;
369  Q3 = Q * Q * Q;
370  A3 = A / 3;
371 
372  if (R * R < Q3) {
373  teta = acos(R / sqrt(Q3));
374  teta3 = teta / 3;
375  q2 = -sqrt(Q) * 2;
376 
377  racines[0].re = q2 * cos(teta3) - A3;
378  racines[0].im = (double) 0.0;
379 
380  racines[1].re = q2 * cos(teta3 + KPI) - A3;
381  racines[1].im = (double) 0.0;
382 
383  racines[2].re = q2 * cos(teta3 - KPI) - A3;
384  racines[2].im = (double) 0.0;
385 
386  } else {
387  if (R >= (double) 0.0)
388  signe = (double) 1.0;
389  else {
390  signe = (double) (-1.0);
391  R = -R;
392  }
393  S =
394  -signe * pow((R + sqrt(R * R - Q3)),
395  (double) (1.0 / 3.0));
396  if (S != (double) 0.0)
397  T = Q / S;
398  else
399  T = (double) 0.0;
400 
401  racines[0].re = S + T - A3;
402  racines[0].im = (double) 0.0;
403 
404  racines[1].re = -(S + T) / 2 - A3;
405  racines[1].im = (S - T) * sqrt((double) 3.0) / 2;
406 
407  racines[2].re = -(S + T) / 2 - A3;
408  racines[2].im = -(S - T) * sqrt((double) 3.0) / 2;
409  }
410 
411  return (racines);
412 }
