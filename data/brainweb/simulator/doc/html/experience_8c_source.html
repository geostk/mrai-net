435 void SetGdechExperience(EXPERIENCE3D *expr,OBJECT3D *object)
436 {
437  int i,j,k;
438  int x,y,z;
439  double decx,decy,decz;
440  float def;
441  x= expr->nfovx;
442  y= expr->nfovy;
443  z= expr->nfovz;
444 
445  /* Allocation */
446  if ((expr->G_deche = (float ***) malloc(sizeof(expr->G_deche[0]) * x)) == NULL) {
447  fprintf(stderr, "Malloc error magstate table\n");
448  exit(1);
449  }
450 
451  for (i = 0; i < x; i++) {
452  if ((expr->G_deche[i] = (float **) malloc(sizeof(expr->G_deche[0][0]) * y)) == NULL) {
453  fprintf(stderr, "Malloc error magstate table \n");
454  exit(1);
455  }
456  }
457  if ((expr->G_deche[0][0] = (float *) malloc(sizeof(expr->G_deche[0][0][0]) * x * y * z)) == NULL) {
458  fprintf(stderr, "Malloc error G_deche table \n");
459  exit(1);
460  }
461  for (i = 0; i < x; i++) for (j = 0; j < y; j++)
462  expr->G_deche[i][j] = expr->G_deche[0][0] + j * z + i * y * z;
463 
464  decx= expr->nfovx/2.;
465  decy= expr->nfovy/2.;
466  decz= expr->nfovz/2.;
467 
468 
469  switch (expr->FLAG_DECH)
470  {
471  case 0: /* No default accounting */
472  {
473  for (i = 0; i < expr->nfovx; i++)
474  for (j = 0; j < expr->nfovy; j++)
475  for (k = 0; k < expr->nfovz; k++)
476  expr->G_deche[i][j][k] = 0.0;
477  break;
478  }
479  case 1: /* Accounting the default linked only to the object */
480  {
481  for (i = 0; i < expr->nfovx; i++)
482  for (j = 0; j < expr->nfovy; j++)
483  for (k = 0; k < expr->nfovz; k++)
484  expr->G_deche[i][j][k] = ((float)expr->B0)* object->fielddefault[k][j][i];
485  break;
486  }
487  case 2: /* Accounting the default linked only to the main magnet B0 with a parabolic form */
488  {
489  for (i = 0; i < expr->nfovx; i++)
490  for (j = 0; j < expr->nfovy; j++)
491  for (k = 0; k < expr->nfovz; k++)
492  expr->G_deche[i][j][k] = ((float)expr->B0)*((i-decx)*(i-decx)+(j-decy)*(j-decy)+(k-decz)*(k-decz))*expr->b0def/(decx*decx+decy*decy+decz*decz);
493  break;
494  }
495  case 3: /* Accounting both effects : case 1 + 2 */
496  {
497  for (i = 0; i < expr->nfovx; i++)
498  for (j = 0; j < expr->nfovy; j++)
499  for (k = 0; k < expr->nfovz; k++)
500  {
501  def =((float)expr->B0)*((i-decx)*(i-decx)+(j-decy)*(j-decy)+(k-decz)*(k-decz))*expr->b0def/(decx*decx+decy*decy+decz*decz);
502  expr->G_deche[i][j][k] = def + ((float)expr->B0)* object->fielddefault[k][j][i];
503  }
504  break;
505  }
506  default: /* No accounting */
507  {
508  for (i = 0; i < expr->nfovx; i++)
509  for (j = 0; j < expr->nfovy; j++)
510  for (k = 0; k < expr->nfovz; k++)
511  expr->G_deche[i][j][k] = 0.0;
512  break;
513  }
514  }

525 void SetB1(EXPERIENCE3D *expr)
526 {
527 // ASCII file B1t.txt must contain a complex 3D map of the normalized transmit field (max module = 1).
528 // ASCII file B1r.txt must contain a complex 3D map of the normalized receive field.
529 // Each line of these files contains 2 floats representing the real and imaginary parts of the B1 value.
530 // The order of the lines is as follows (x, y and z vary from 0 to sizeX, sizeY, sizeZ, e.g. 64 x 64 x 40):
531 //
532 // 1st line corresponds to pixel (x=0, y=0, z=0) in volume corner
533 // 2nd line (x=1, y=0, z=0)
534 // ...
535 // 64-th line (x=63, y=0, z=0)
536 // 65-th line (x=0, y=1, z=0)
537 // 66-th line (x=1, y=1, z=0)
538 // ...
539 // 4096-th line (x=63, y=63, z=0)
540 // 4097-th line (x=0, y=0, z=1)
541 // ...
542 // last line (x=63, y=63, z=39) in opposite volume corner
543 //
544 // If ever the object to be imaged has a cylindrical symmetry, B1r does not need to be provided in addition to B1t:
545 // a mirror symmetry is then used to find B1r from B1t, i.e.: Re(B1r(x,y)) = Im(B1t(y,x)) and Im(B1r(x,y)) = Re(B1t(y,x))
546  char B1t_File[80] = "../data/B1map/B1t.txt"; // transmit field
547  char B1r_File[80] = "../data/B1map/B1r.txt"; // reception field
548  FILE *b1t_file, *b1r_file;
549  int i,j,k;
550  int x,y,z;
551  x = expr->ntx;
552  y = expr->nty;
553  z = expr->ntz;
554 
555  printf("x = %i, y = %i, z = %i\n", x, y, z);
556 
557  b1t_file = fopen(B1t_File, "r");
558  if (b1t_file == NULL) {
559  printf("Unable to read the B1t(r) file %s \n", B1t_File);
560  exit(0);
561  }
562  b1r_file = fopen(B1r_File, "r");
563 
564  /* B1 Allocation*/
565 
566  expr->B1t = (PPPVOLUME_COMPLEX_FLOAT) IdVolAlloc(x,y,z,VOL_COMPLEX_FLOAT);
567  if (!expr->B1t)
568  {
569  printf("Impossible d'allouer B1t(r) ! \n");
570  exit(0);
571  }
572  expr->B1r = (PPPVOLUME_COMPLEX_FLOAT) IdVolAlloc(x,y,z,VOL_COMPLEX_FLOAT);
573  if (!expr->B1r)
574  {
575  printf("Impossible d'allouer B1r(r) ! \n");
576  exit(0);
577  }
578 
579 // Set B1(r) to default values
580 
581  printf("B1 File reading %s\n", B1t_File);
582  for (k = 0; k < z; k++) for (j = 0; j < y; j++) for (i = 0; i < x; i++){
583  fscanf(b1t_file, "%f %f", &expr->B1t[k][j][i].re, &expr->B1t[k][j][i].im);
584  }
585  fclose(b1t_file);
586 
587  if (b1r_file == NULL) {
588  printf("Unable to read the B1r(r) file %s \n", B1r_File);
589  printf("Assume cylindrical symmetry of the phantom/subject, i.e.: Re(B1r(x,y)) = Im(B1t(y,x)) and Im(B1r(x,y)) = Re(B1t(y,x))\n");
590  for (k = 0; k < z; k++) for (j = 0; j < y; j++) for (i = 0; i < x; i++){
591  expr->B1r[k][i][j].re = expr->B1t[k][j][i].im;
592  expr->B1r[k][i][j].im = expr->B1t[k][j][i].re;
593  }
594  }
595  else {
596  printf("Lecture du fichier %s\n", B1r_File);
597  for (k = 0; k < z; k++) for (j = 0; j < y; j++) for (i = 0; i < x; i++){
598  fscanf(b1r_file, "%f %f", &expr->B1r[k][j][i].re, &expr->B1r[k][j][i].im);
599  }
600  fclose(b1r_file);
601  }
602 }

662 void SetObjectExperienceComponent(EXPERIENCE3D *expr, OBJECT3D *object,int component,int FLAG_DECH)
663 {
664  if ((object->nbcomponent>2)||(object->nbparam>4))
665  {
666  printf("This kind of object is not manageable at the moment!\n");
667  exit(0);
668  }
669  /* Chemical shift in rad/s */
670 
671  if (component==0)
672  expr->G_edw = object->chemshift[WATER] * 2. * M_PI ;
673  else if (component==1)
674  expr->G_edw = object->chemshift[FAT] * 2. * M_PI;
675  else
676  {
677  printf("Component not defined !\n");
678  exit(1);
679  }
680 
681  expr->FLAG_DECH = FLAG_DECH ; /* Field default accounting */
682 
683  /* Fov and size */
684  /* By default, the fov equals the object size */
685  expr->lx = object->lx;
686  expr->ly = object->ly;
687  expr->lz = object->lz;
688  expr->fovx = object->lx;
689  expr->fovy = object->ly;
690  expr->fovz = object->lz;
691 
692  /* Number of object point */
693  expr->nfovx=object->x;
694  expr->nfovy=object->y;
695  expr->nfovz=object->z;
696 
697  /* The step px,py,pz (used by the computation kernel to compute dw) are
