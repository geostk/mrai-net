109 char * IdStrCreateNewSuffix (char * nom, char * suffix )
110 {
111  char * fin;
112  char * nouveau;
113  char * ajout;
114  int car_copie;
115  int taille;
116 
117  /* fin pointe sur fin du nom **/
118  for ( fin = nom; *fin != '\0'; fin ++ );
119  /* cherche 1 '.' avant la fin */
120  for ( ajout = fin; (*ajout!='.'); ajout -- ) {
121  /* Pas de '.' trouve **********/
122  if ( ajout <= nom ) {
123  ajout = fin ;
124  break;
125  }
126  /* carac. interdit trouve *****/
127  if (
128 
129 #if defined(_MSC_VER)
130  (*ajout==':')||(*ajout=='\\')
131 #else
132  (*ajout=='/')
133 #endif
134 
135  ) {
136  ajout = fin ;
137  break;
138  }
139  }
140  /* calcul taille finale *******/
141  taille = (int) (ajout-nom) + 2 + strlen(suffix) ;
142  /* et reservation memoire *****/
143  nouveau = (char *)malloc (taille) ;
144  /* copie du prefixe ***********/
145  for ( car_copie = 0; nom < ajout; car_copie ++ )
146  nouveau[car_copie] = *(nom++);
147  /* et ajout du suffixe ********/
148  for ( ajout = suffix; car_copie < taille; car_copie++ )
149  nouveau[car_copie] = *(ajout++);
150 
151  return nouveau;
