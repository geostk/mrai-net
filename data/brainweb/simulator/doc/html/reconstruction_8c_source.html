53 void VolRFFiltering(PPPVOLUME_COMPLEX_DOUBLE volrf,int type)
54 {
55  int i,j,k;
56  double x,y,z;
57  double w,d;
58 
59  x= (double) IdVolDimX(volrf);
60  y= (double) IdVolDimY(volrf);
61  z= (double) IdVolDimZ(volrf);
62 
63 switch(type)
64 {
65 case 0: /* Hanning */
66  {
67  if ((x!=1)&&(y!=1)&&(z!=1)) /*3D*/
68  {
69  for(k=0;k<z;k++) for(j=0;j<y;j++) for(i=0;i<x;i++)
70  {
71  d=sqrt((i-x/2.)*(i-x/2.)/x/x*4+(j-y/2.)*(j-y/2.)/y/y*4+(k-z/2)*(k-z/2.)/z/z*4);
72  if (d>1.0) d=1.0;
73  w=0.5 + 0.5*cos(M_PI*d);
74  volrf[k][j][i].re=volrf[k][j][i].re * w;
75  volrf[k][j][i].im=volrf[k][j][i].im * w;
76  }
77  }
78  if ((x!=1)&&(y!=1)&&(z==1)) /*2D*/
79  {
80  for(k=0;k<z;k++) for(j=0;j<y;j++) for(i=0;i<x;i++)
81  {
82  d=sqrt((i-x/2.)*(i-x/2.)/x/x*4+(j-y/2.)*(j-y/2.)/y/y*4);
83  if (d>1.0) d=1.0;
84  w=0.5 + 0.5*cos(M_PI*d);
85  volrf[k][j][i].re=volrf[k][j][i].re * w;
86  volrf[k][j][i].im=volrf[k][j][i].im * w;
87  }
88  }
89  if ((x!=1)&&(y==1)&&(z==1)) /*1D*/
90  {
91  for(k=0;k<z;k++) for(j=0;j<y;j++) for(i=0;i<x;i++)
92  {
93  d=sqrt((i-x/2.)*(i-x/2.)/x/x*4);
94  w=0.5 + 0.5*cos(M_PI*d);
95  volrf[k][j][i].re=volrf[k][j][i].re * w;
96  volrf[k][j][i].im=volrf[k][j][i].im * w;
97  }
98  }
99  } /* End Hanning */
100 case 2:
101  {
102  }
103 case 3:
104  {
105  }
106 } /* End switch */
107 }

117 PPPVOLUME_COMPLEX_DOUBLE RecVolIFFTComplexeFromVol (PPPVOLUME_COMPLEX_DOUBLE volrf)
118 {
119  PPPVOLUME_COMPLEX_DOUBLE volrec;
120  int i,j,k;
121  int x,y,z;
122 
123  volrf= (PPPVOLUME_COMPLEX_DOUBLE) IdVolIFFT((PPPVOLUME_COMPLEX) volrf);
124  /* Careful : volrf output of IFFT is different from the volrf input. Thus it is desalloacted in this function */
125 
126  /* x permutation of the IFFT */
127  x=IdVolDimX(volrf);
128  y=IdVolDimY(volrf);
129  z=IdVolDimZ(volrf);
130  volrec = (PPPVOLUME_COMPLEX_DOUBLE) IdVolAlloc(x,y,z,VOL_COMPLEX_DOUBLE);
131  if(!volrec)
132  {
133  printf("Impossible d'allouer un volume dans RecVolIFFTComplexe !\n");
134  exit(0);
135  }
136  for(k=0;k<z;k++) for(j=0;j<y;j++) for(i=0;i<x;i++)
137  {
138  volrec[k][j][i].re=volrf[k][j][x-1-i].re;
139  volrec[k][j][i].im=volrf[k][j][x-1-i].im;
140  }
141 
142  IdVolFree(volrf);
143  return(volrec);
144 }

154 PPPVOLUME_FLOAT RecVolIFFTModuleFromVol (PPPVOLUME_COMPLEX_DOUBLE volrf)
155 {
156  PPPVOLUME_COMPLEX_DOUBLE volrec;
157  PPPVOLUME_FLOAT module;
158  int i,j,k;
159  int it,jt,kt;
160  int x,y,z;
161  double a,b;
162 
163  volrec= (PPPVOLUME_COMPLEX_DOUBLE) RecVolIFFTComplexeFromVol(volrf);
164  x = IdVolDimX(volrec); y=IdVolDimY(volrec); z=IdVolDimZ(volrec);
165 
166  module = (PPPVOLUME_FLOAT) IdVolAlloc(x,y,z,VOL_FLOAT);
167 
168  /* Module computation with dimx*dimy*dimz permutations */
169  for(k=0;k<z;k++) for(j=0;j<y;j++) for(i=0;i<x;i++)
170  {
171  it = (i + (x / 2)) % x;
172  jt = (j + (y / 2)) % y;
173  kt = (k + (z / 2)) % z;
174  a = volrec[k][j][i].re;
175  b = volrec[k][j][i].im;
176  module[kt][jt][it] = (float) sqrt((a*a) + (b*b));
177  }
178 
179  IdVolFree(volrec);
180 
181  return(module);
182 }
