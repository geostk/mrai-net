326 OBJECT3D* AllocObject(int x, int y, int z, int nbcomp, int nbparam)
327 {
328  OBJECT3D *object;
329  int i, j, k, t;
330  int nbvol;
331 
332  nbvol = nbcomp*nbparam;
333 
334  object = (OBJECT3D *)malloc(sizeof(OBJECT3D));
335  if (!object)
336  {
337  printf("Unable to allocate an OBJECT3D ! \n");
338  exit(0);
339  }
340 
341  object->values = (PPPPSEQUENCE_USHORT)IdSeqAlloc(x, y, z, nbvol, SEQ_USHORT);
342  if (!object->values)
343  {
344  printf("Unable to allocate a SEQUENCE ! \n");
345  exit(0);
346  }
347 
348  for (t = 0; t < nbvol; t++)
349  for (i = 0; i < x; i++) for (j = 0; j < y; j++) for (k = 0; k < z; k++)
350  {
351  object->values[t][k][j][i] = 0;
352  }
353 
354  object->chemshift = (PSIGNAL_FLOAT)IdSigAlloc(nbcomp, SIG_FLOAT);
355  if (!object->chemshift)
356  {
357  printf("Unable to allocate SIGNAL! \n");
358  exit(0);
359  }
360  for (t = 0; t < nbcomp; t++)
361  {
362  object->chemshift[t] = 0.0;
363  }
364 
365  object->deltaB0 = (PPPVOLUME_FLOAT)IdVolAlloc(x, y, z, VOL_FLOAT);
366  if (!object->deltaB0)
367  {
368  printf("Unable to allocate VOLUME! \n");
369  exit(0);
370  }
371 
372  for (i = 0; i < x; i++) for (j = 0; j < y; j++) for (k = 0; k < z; k++)
373  {
374  object->deltaB0[k][j][i] = 0.0;
375  }
376 
377  object->fielddefault = (PPPVOLUME_FLOAT)IdVolAlloc(x, y, z, VOL_FLOAT);
378  if (!object->fielddefault)
379  {
380  printf("Unable to allocate VOLUME! \n");
381  exit(0);
382  }
383 
384  for (i = 0; i < x; i++) for (j = 0; j < y; j++) for (k = 0; k < z; k++)
385  {
386  object->fielddefault[k][j][i] = 0.0;
387  }
388 
389  object->x = x; object->y = y; object->z = z;
390  object->nbcomponent = nbcomp;
391  object->nbparam = nbparam;
392  object->component = 0; /* Default active component = 0 */
393 
394  /* We give to the object a size of 1x1x1 meter */
395  object->lx = 1.0; object->ly = 1.0; object->lz = 1.0;
396 
397  return(object);
398 }

520 OBJECT3D* ReadObject(char * name)
521 {
522  PPPPSEQUENCE_USHORT seq;
523  OBJECT3D * object;
524  int x, y, z, nbparam, nbcomp, nbvol;
525  int t, i, j, k;
526  float lx, ly, lz;
527 
528  seq = (PPPPSEQUENCE_USHORT)IdAcrReadFile(name, SEQ_USHORT);
529  x = IdSeqDimX(seq);
530  y = IdSeqDimY(seq);
531  z = IdSeqDimZ(seq);
532  nbvol = IdSeqDimT(seq);
533 
534  if (nbvol > 3)
535  {
536  nbparam = 3; /* Default value : ro,t1,t2 */
537  nbcomp = 2; /* Default value : 2, WATER and FAT*/
538  }
539  else
540  {
541  nbparam = 3; /* Default value : ro,t1,t2 */
542  nbcomp = 1; /* Default value : 1, WATER */
543  }
544  lx = 1.0; ly = 1.0; lz = 1.0;
545 
546  object = (OBJECT3D *)AllocObject(x, y, z, nbcomp, nbparam);
547  for (t = 0; t < nbvol; t++)
548  for (i = 0; i < x; i++) for (j = 0; j < y; j++) for (k = 0; k < z; k++)
549  {
550  /* Not very efficient, better possible using memcpy ! */
551  object->values[t][k][j][i] = seq[t][k][j][i];
552  }
553  IdSeqFree(seq);
554 
555  object->nbcomponent = nbcomp;
556  object->nbparam = nbparam;
557  object->lx = lx; object->ly = ly; object->lz = lz;
558  object->x = x; object->y = y; object->z = z;
559 
560  return(object);
561 }
