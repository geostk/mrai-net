101 OBJECT3D* LabelToObject(PPPVOLUME_UCHAR label, PPIMAGE_FLOAT value)
102 {
103  int x, y, z;

105  OBJECT3D *object;
106  int i, j, k;
107 
108  x = IdVolDimX(label);
109  y = IdVolDimY(label);
110  z = IdVolDimZ(label);
111 
112  object = AllocObject(x, y, z, 1, 3);
113 
114  for (k = 0; k < z; k++) for (j = 0; j < y; j++) for (i = 0; i < x; i++)
115  {
116  lab = label[k][j][i];
117  ro = (int)(value[lab][0] + RandGaussian(value[lab][1], 0.0, 0));
118  t1 = (int)(value[lab][2] + RandGaussian(value[lab][3], 0.0, 0));
119  t2 = (int)(value[lab][4] + RandGaussian(value[lab][5], 0.0, 0));
120  SetObjectPoint(object, i, j, k, ro, t1, t2, WATER);
121  }
122  return object;
123 }
