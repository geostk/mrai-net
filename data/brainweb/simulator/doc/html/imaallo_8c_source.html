221 PPIMAGE IdImaAlloc(int sx, // Number of columns (width)
222  int sy, // Number of lignes (height)
223  int ty)
224 {
225  PRIVATE_IMAGE *si;
226  PPIMAGE pi = NULL;
227  if (!(ty & IDO_TYPE) == IMA) {
228  IdErrPrintf
229  ("Allocation impossible. Objet de type incompatible(%x)\n",
230  ty);
231  IdErrno = IDERR_WRONG_TYPE;
232  return (PPIMAGE) 0;
233  }
234  si =
235  (PRIVATE_IMAGE *) calloc(1, sizeof(PRIVATE_IMAGE) +
236  sy * sizeof(void *));
237  if (si) {
238  si->DimX = sx;
239  si->DimY = sy;
240  si->Type = ty;
241  si->subObject = 0;
242  si->UsedNbX = 0;
243  si->UsedNbY = 0;
244  si->_fichier = 0;
245  si->_message = 0;
246  pi = (PPIMAGE) & (si[1]);
247  if (ty == IMA_BIT) {
248  sx = (sx + 7) >> 3;
249  ty = IMA_UCHAR;
250  }
251  if (!_IdAllocLinesOfPlane(pi, sx, sy, IdSizeOfType(pi))) {
252  IdImaFree(pi);
253  IdErrPrintf
254  ("Echec d'allocation d'image. Memoire saturee.");
255  IdErrno = IDERR_ALLOC_MEM;
256  return (PPIMAGE) 0;
257  }
258  } else {
259  IdErrPrintf
260  ("Echec d'allocation PRIVATE_IMAGE. Memoire saturee.");
261  }
262  return (pi);
263 }
