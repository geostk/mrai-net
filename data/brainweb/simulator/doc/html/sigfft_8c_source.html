56 IdSigFFT(PSIGNAL_COMPLEX sig, int isign)
57 {
58  int n, nn, mmax, m, j, istep, i;
59  REAL wtemp, wr, wpr, wpi, wi, theta;
60  REAL tempr, tempi;
61 
62  if (!sig) {
63  IdErrPrintf("IdSigFFT: Parametrer is NULL\n");
64  return;
65  }
66 
67  if (IdLibidoType(sig) != SIG) {
68  IdErrPrintf("IdSigFFT: Parametrer is NOT a SIGNAL\n");
69  return;
70  }
71  if ((IdLibidoDataType(sig) != TY_COMPLEX) &&
72  (IdLibidoDataType(sig) != TY_COMPLEX_DOUBLE))
73  {
74  IdErrPrintf
75  ("IdSigFFT: Parametrer is NOT a PSIGNAL_COMPLEX_DOUBLE\n");
76  return;
77  }
78 
79  nn = IdSigDimX(sig);
80  n = nn;
81  j = 1;
82  for (i = 1; i < n; i++) {
83  if (j > i) {
84  SWAP(sig[j - 1].re, sig[i - 1].re);
85  SWAP(sig[j - 1].im, sig[i - 1].im);
86  }
87  m = n >> 1;
88  while (m >= 2 && j > m) {
89  j -= m;
90  m >>= 1;
91  }
92  j += m;
93  }
94  mmax = 1;
95  while (n > mmax) {
96  istep = 2 * mmax;
97  theta = 6.28318530717959 / (-isign * mmax * 2);
98  /*
99  * recipes in C
100  * +isign
101  */
102  wtemp = sin(0.5 * theta);
103  wpr = -2.0 * wtemp * wtemp;
104  wpi = sin(theta);
105  wr = 1.0;
106  wi = 0.0;
107  for (m = 0; m < mmax; m++) {
108  for (i = m; i < n; i += istep) {
109  j = i + mmax;
110  tempr = wr * sig[j].re - wi * sig[j].im;
111  tempi = wr * sig[j].im + wi * sig[j].re;
112  sig[j].re = sig[i].re - tempr;
113  sig[j].im = sig[i].im - tempi;
114  sig[i].re += tempr;
115  sig[i].im += tempi;
116  }
117  wtemp = wr;
118  wr = (wtemp) * wpr - wi * wpi + wr;
119  wi = wi * wpr + wtemp * wpi + wi;
120  }
121  mmax = istep;
122  }
123  if (isign == -1)
124  for (i = 0; i < n; i++) {
125  sig[i].re /= nn;
126  sig[i].im /= nn;
127  }
128 }
