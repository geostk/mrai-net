218 OBJECT3D * CreateObject3DSphereKi(int dimx, int dimy, int dimz)
219 {
220  OBJECT3D * objet;
221  int nbcomp,nbparam;
222  int i,j,k;
223  PPPVOLUME_FLOAT ro,t1,t2,ki,dech;
224  PVECTOR_FLOAT roval,t1val,t2val,kival;
225  char filelabel[80],filedech[80];
226  filelabel[0]='\0';
227  filedech[0]='\0';
228 
229  strcpy(filelabel,"hightresolution.vol");
230  strcpy(filedech,"Defchamps.vol");
231 
232  nbcomp = 1;
233  nbparam= 4; /* ro, t1, t2 ,db */
234  objet = AllocObject(dimx,dimy,dimz,nbcomp,nbparam);
235 
236  roval=AllocInitVectFloat(2);
237  roval[0]=100; /* region 0 */
238  roval[1]=90; /* region 1 */
239  ro=GetParamFromDataFile(filelabel,roval);
240 
241  t1val=AllocInitVectFloat(2);
242  t1val[0]=90 ; /* region 0 */
243  t1val[1]=400; /* region 1 */
244  t1=GetParamFromDataFile(filelabel,t1val);
245 
246  t2val=AllocInitVectFloat(2);
247  t2val[0]=40 ; /* region 0 */
248  t2val[1]=50; /* region 1 */
249  t2=GetParamFromDataFile(filelabel,t2val);
250 
251  kival=AllocInitVectFloat(2);
252  kival[0]=0; /* region 0 */
253  kival[1]=-9.05e-6; /* region 1 */
254  ki=GetParamFromDataFile(filelabel,kival);
255 
256  for (i=0;i<dimx;i++) for(j=0;j<dimy;j++) for(k=0;k<dimz;k++)
257  SetObjectPointSusceptibility(objet,i,j,k,(int)ro[k][j][i],(int)t1[k][j][i],(int)t2[k][j][i],(int)ki[k][j][i]);
258 
259 /* Reading the field default values in a file
260  * This file has been obtained with one of the
261  * suceptibility default computation method */
262  if (!strcmp(filedech,NULL))
263  {
264  dech = (PPPVOLUME_FLOAT) IdAcrReadFile(filedech,VOL_FLOAT);
265  SetObjectFieldDefault3D(objet,dech,dimx,dimy,dimz);
266  }
267  else
268  {
269  printf("Unable to read the field default values %s \n",filedech);
270  exit(1);
271  }
272 
273  IdVolFree(ro);
274  IdVolFree(t1);
275  IdVolFree(t2);
276  IdVolFree(ki);
277  IdVolFree(dech);
278  return(objet);
279 }
