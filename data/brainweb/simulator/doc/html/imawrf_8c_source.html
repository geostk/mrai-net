85 int IdImaWriteRawFile(
86  char *nomfic,
87  PPIMAGE im)
88 {
89  int i;
90  FILE *fp;
91  if(!im) { IdErrno=1; return(1);}
92  fp=fopen(nomfic,ID_WFILE_BIN);
93  if(!fp) { IdErrno=2; return(1);}
94  for(i=0;i<IdImaDimY(im);i++)
95  fwrite(im[i],IdSizeOfType(im),IdImaDimX(im),fp);
96  fclose(fp);
97  return(0);
98 }
