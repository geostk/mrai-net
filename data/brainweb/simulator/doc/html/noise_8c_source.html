51 double RandGaussian(double s, double m, int initseed)
52 {
53  double r1,r2;
54  double x;
55 
56  if (initseed!=0) srand((unsigned)(initseed*time(NULL)));
57 
58  r1=rand()/(double)RAND_MAX;
59  r2=rand()/(double)RAND_MAX;
60 
61  x= (s*sqrt(-2.0*log(r1))*cos(2*PI*r2)) + m;
62 
63  return(x);

101  for(k=0;k<z;k++) for(j=0;j<y;j++)
