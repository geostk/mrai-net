95 {
96  PLIST_ELEMENT e, p;
97 
98  if (!list) {
99  IdErrno = IDERR_POINTER_IS_NULL;
100  return (0);
101  }
102 
103  e = calloc(1, sizeof(LIST_ELEMENT));
104  if (!e) {
105  IdErrno = IDERR_ALLOC_ELEM;
106  return (0);
107  }
108 
109  (_IdLstPrivate(list)->NbElems)++;
110  e->Numero = IdLstNbElems(list);
111  switch (mode & 0x0f) {
112  case LST_BEG:
113  if (mode & (LST_NEXT | LST_BIDIR)) {
114  p = IdLstFirst(list);
115  if (p == 0)
116  IdLstLast(list) = e;
117  e->Next = p;
118  IdLstFirst(list) = e;
119  e->Object = obj;
120  if (mode & LST_BIDIR)
121  if (e->Next)
122  e->Next->Pred = e;
123  }
124  if (mode & LST_PRED) {
125  e->Pred = IdLstFirst(list);
126  IdLstFirst(list) = e;
127  e->Object = obj;
128  }
129  break;
130  case LST_END:
131  if (mode & (LST_NEXT | LST_BIDIR)) {
132 
133  p = IdLstFirst(list);
134  if (p) {
135  while (p->Next)
136  p = p->Next;
137  p->Next = e;
138  if (mode & LST_BIDIR)
139  e->Pred = p;
140  } else
141  IdLstFirst(list) = e;
142  e->Object = obj;
143  IdLstLast(list) = e;
144  }
145  if (mode & LST_PRED) {
146  p = IdLstFirst(list);
147  if (p) {
148  while (p->Pred)
149  p = p->Pred;
150  p->Pred = e;
151  } else
152  IdLstFirst(list) = e;
153  e->Object = obj;
154  }
155  break;
156  case LST_ELEM:
157  if (mode & (LST_NEXT | LST_BIDIR)) {
158  if (elem) {
159  if (elem == IdLstLast(list))
160  IdLstLast(list) = e;
161  e->Next = elem->Next;
162  elem->Next = e;
163  if (mode & LST_BIDIR) {
164  e->Pred = elem;
165  if (e->Next)
166  e->Next->Pred = e;
167  }
168  }
169  e->Object = obj;
170  }
171  if (mode & LST_PRED) {
172  if (elem) {
173  e->Pred = elem->Pred;
174  elem->Pred = e;
175  }
176  e->Object = obj;
177  }
178  break;
179 
180  case LST_ELEM_AV:
181  if (mode & (LST_NEXT | LST_BIDIR)) {
182  if (elem) {
183  if (elem == IdLstFirst(list))
184  IdLstFirst(list) = e;
