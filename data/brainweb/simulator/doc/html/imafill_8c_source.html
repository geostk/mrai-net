54 RESUME: Rempli les trous ds les objets connexes d'1 image binaire
55 
56 DESCRIPTION:
57  Remplit les trous dans les objets connexes d'une image binaire
58  (Objets blancs sur fond noir).
59  A noter: un cadre noir de largeur 1 pixel remplace le bord de l'image.
60  PLUS MAINTENANT ! ...
61 
62  ATTENTION : L'image source etait modifiee, ainsi que l'image destination.
63  Ce n'etait pas precise dans la Doc, c'etait peut-etre un Bug....
64  Ce n'est plus le cas maintenant (21/4/99)
65 
66 SYNTAXE: int retCode=IdImaFill (PPIMAGE_UCHAR imSource, PPIMAGE_UCHAR imDest, int connexite);
67 
68 RETOUR: type : int
69  role : -1 si echec, sinon le nombre de trous remplis.
70 
71 PARAMETRES: nom : ims
72  type : PPIMAGE_UCHAR
73  role : Pointeur vers l'image source
74 
75  nom : imd
76  type : PPIMAGE_UCHAR
77  role : Pointeur vers l'image resultat
78  Si ce pointeur est nul, l'image est allouee par
79  la fonction. NON CE POINTEUR NE PEUT PAS ETRE NULL MW
80 
81  nom : connexite
82  type : int
83  role : connexite des objets:
84  connexite = 1 : 8-connexite
85  connexite = 0 : 4-connexite
86 
87 FICHIER: imafill.c
88 
89 EXEMPLE: if ( (status=IdImaFill(ims,imd,1)) == -1 )
90  IdErrPrintf("ERREUR: %s",IdErrMsg(IdErrno));

93 int
94 IdImaFill(ims,imd,connexite)
95 PPIMAGE_UCHAR ims,imd;
96 int connexite;
97 {
98 PPIMAGE_SHORT imo;
99 PPIMAGE_UCHAR imX,imY;
100 FILE *fich_res,*fich_mod;
101 char nom_res[100],nom_mod[100];
102 int nb,surf_min=1,x,y;
103 short val;
104 unsigned char tval;
105  /* Controle des l'images **/
106  if ( !ims || IdImaType(ims)!=IMA_UCHAR ){
107  IdErrno = IDERR_WRONG_TYPE;
108  return -1;
109  }
110  if ( !imd || IdImaType(imd)!=IMA_UCHAR ){
111  IdErrno = IDERR_WRONG_TYPE;
112  return -1;
113  }
114  if ( (connexite != 0) && (connexite != 1) ){
115  IdErrno = IDERR_INVALID_NUMBER;
116  return -1;
117  }
118 
119 /* On sauve le cadre de largeur un pixel autour de l'image *****/
120 /* On repousse vers l'interieur les pixels blancs qui touchent le bord *****/
121 /* On incruste un cadre noir de largeur un pixel autour de l'image *****/
122 
123 imX=(PPIMAGE_UCHAR)IdImaAlloc(IdImaDimX(ims),2,IMA_UCHAR);
124 imY=(PPIMAGE_UCHAR)IdImaAlloc(2,IdImaDimY(ims),IMA_UCHAR);
125 
126  for(x=0;x<IdImaDimX(ims);x++)
127  { imX[0][x] = ims[0][x];
128  imX[1][x] = ims[IdImaDimY(ims)-1][x];
129 
130  if (ims[0][x]==255)
131  {
132  ims[1][x]=255;
133  ims[0][x]=0;
134  }
135  if (ims[IdImaDimY(ims)-1][x]==255)
136  {
137  ims[IdImaDimY(ims)-2][x]=255;
138  ims[IdImaDimY(ims)-1][x]=0;
139  }
140  }
141 
142  for(y=1;y<IdImaDimY(ims)-1;y++)
143  { imY[y][0] = ims[y][0];
144  imY[y][1] = ims[y][IdImaDimX(ims)-1];
145 
146  if (ims[y][0]==255)
147  {
148  ims[y][1]=255;
149  ims[y][0]=0;
150  }
151  if (ims[y][IdImaDimX(ims)-1]==255)
152  {
153  ims[y][IdImaDimX(ims)-2]=255;
154  ims[y][IdImaDimX(ims)-1]=0;
155  }
156  }
157 
158 /***** Transformation de l'image en negatif *****/
159  for(y=0;y<IdImaDimY(ims);y++)
160  for(x=0;x<IdImaDimX(ims);x++){
161  tval=ims[y][x];
162  ims[y][x]= ~tval;
163  }
164 
165 /***** Allocation de l'image label et initialisation a zero *****/
166  imo=(PPIMAGE_SHORT) IdImaAlloc(IdImaDimX(ims),IdImaDimY(ims),IMA_SHORT);
167  if ( !imo ) {
168  IdErrno = IDERR_ALLOC_MEM;
169  return -1;
170  }
171  for(y=0;y<IdImaDimY(imo);y++)
172  for(x=0;x<IdImaDimX(imo);x++)
173  imo[y][x]=0;
174 
175 /***** Recherche des objets (trous) *****/
176 
177 /* ! ! ! Nom de fichiers fixes, 1 seul appel a la fois ! ! ! */
178 
179 #if ( _SUNC_ )
180  strcpy(nom_mod,getenv("HOME")); strcat(nom_mod,"/tmp/ftemp.mod");
181  strcpy(nom_res,getenv("HOME")); strcat(nom_res,"/tmp/ftemp.res");
182 #else
183  strcpy(nom_mod,"ftemp.mod");
184  strcpy(nom_res,"ftemp.res");
185 #endif
186  if(! (fich_mod=fopen(nom_mod,ID_WFILE_TEXT)) ) {
187  IdErrno = IDERR_OPEN_FAILED;
188  IdErrPrintf("ERREUR: %s: %s\n",IdErrMsg(IdErrno),nom_mod);
189  return -1;
190  }
191  if(! (fich_res=fopen(nom_mod,ID_WFILE_TEXT)) ) {
192  IdErrno = IDERR_OPEN_FAILED;
193  IdErrPrintf("ERREUR: %s: %s\n",IdErrMsg(IdErrno),nom_res);
194  return -1;
195  }
196 
197  nb=IdImaLabel(ims,imo,surf_min,fich_res,fich_mod,1-connexite);
198 
199 /* Les connexites des objets et des trous sont complementaires (4 - 8) */
200 
201  if(nb == -1) IdErrPrintf("ERREUR: %s\n",IdErrMsg(IdErrno));
202  fclose(fich_res);fclose(fich_mod);
203 
204 /***** Remplissage des trous *****/
205 
206  for(y=0;y<IdImaDimY(imd);y++)
207  for(x=0;x<IdImaDimX(imd);x++)
208  {
209  val=imo[y][x];
210  if (val==0) imd[y][x]=0; /* val=0 : objets */
211  else if (val<nb) imd[y][x]=0; /* 0<val<nb : trous */
212  else imd[y][x]=255; /* val=nb : fond */
213  }
214 
215 /***** Liberation memoire *****/
216  IdImaFree(imo);
217 
218 /***** Retour en normal negatif de imd *****/
219 
220  for(y=0;y<IdImaDimY(imd);y++)
221  for(x=0;x<IdImaDimX(imd);x++){
222  tval=imd[y][x];
223  imd[y][x]= ~tval;
224 /* MW */
225 /* tval=ims[y][x];
226  ims[y][x]= ~tval;
227 */
228  }
229 
230 if (imd!=ims) /* On ne restaure l'image Source QUE si elle est differente */
231  /* de l'image destination ... JPRx */
232  for(y=0;y<IdImaDimY(imd);y++)
233  for(x=0;x<IdImaDimX(imd);x++){
234  tval=ims[y][x];
235  ims[y][x]= ~tval;
236  }
237 
238 /* On restaure le cadre de largeur un pixel autour de l'image *****/
239 
240  for(x=0;x<IdImaDimX(ims);x++)
241  { imd[0][x] = imX[0][x];
242  imd[IdImaDimY(ims)-1][x] = imX[1][x];
243 
244  ims[0][x] = imX[0][x];
245  ims[IdImaDimY(ims)-1][x] = imX[1][x];
246  }
247 
248  for(y=1;y<IdImaDimY(ims)-1;y++)
249  { imd[y][0] = imY[y][0];
250  imd[y][IdImaDimX(ims)-1] = imY[y][1];
251 
252  ims[y][0] = imY[y][0];
253  ims[y][IdImaDimX(ims)-1] = imY[y][1];
254  }
255 
256 IdImaFree(imX);
257 IdImaFree(imY);
258 
259 return nb ;
260 }
