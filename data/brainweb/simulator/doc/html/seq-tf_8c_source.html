5 This sequence chains the following events
6 - RF pulse
7 - Waiting
8 - Phase Gradient
9 - Waiting

11 - Acquisition with a positive gradient
12 - Rewind of all the gradients (phase and readout)

134 void SeqTrueFisp1D_0(SEQPARAM *seqparam,EXPERIENCE3D* expr)
135 {
136  int i,ntx,phase,altphase;
137  double fovx;
138  double tr,tacq,timp,acq_delay,tgrad,twait; /* ms */
139  double gx;
140  double teta; /* degree */
141  int npy, npz;
142  EVENT* entre, entre_r;
143  entre = &entre_r;
144 
145  printf("Starting SeqTrueFisp1D \n");
146 
147  ntx = expr->ntx;
148  tacq = expr->readout_time*1.e+3;
149  fovx = expr->fovx;
150 
151  tr = seqparam->tr;
152  teta = seqparam->teta;
153 
154  CheckTimesTF(tacq,tr/2.,tr);
155  timp=0.3; /*RF pulse duration in ms */
156  acq_delay = 0;
157  tgrad=(tacq)/2.;
158  twait=(tr-2*tgrad-tacq)/2;
159 
160  gx=(ntx-1)/tacq/G_kHz_G/fovx;
161  npy=0;npz=0;
162 
163  SetSpoilingFlag(entre,INACTIVE); /* No XY Spoiling after acquisition for true fisp*/
164  DoWaiting(expr,entre,ATT_INIT); /* To get an initial magnetisation M0 */
165 
166  altphase=-1;/* Phase alternance at each acquisition */
167  phase = 1;
168  /* Iterated excitations to get an equilibrium state */
169  for (i = 0; i<10*ntx; i++)
170  {
171  DoPulseRect(expr,entre,teta * phase,timp);
172  DoWaiting(expr,entre,tr);
173  phase= phase * altphase;
174  }
175  ResetToExperience(expr);
176  DoPulseRect(expr,entre,teta * phase,timp);
177  DoWaiting(expr,entre,twait);
178  DoGradient( expr,entre,tgrad,-gx,0.0,0.0);
179  DoAcqFrequenceX(expr,entre, +gx, npy, npz,acq_delay, phase);
180 
181  printf("Ending SeqTrueFisp1D \n");
182 }

240  {

249  DoWaiting(expr,entre,twait);
