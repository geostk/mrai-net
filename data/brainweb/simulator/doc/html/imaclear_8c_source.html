70 PPIMAGE IdImaClear ( PPIMAGE image )
71 {
72  int y;
73  int taille_ligne;
74  /* Controle de l'image **/
75  if ( !image ) { /* Alloc. si necessaire ***/
76  IdErrno = IDERR_POINTER_IS_NULL;
77  return 0;
78  }else if ( IdLibidoType(image) != IMA) {
79  IdErrno = IDERR_WRONG_TYPE;
80  return 0;
81  }
82  taille_ligne = (IdImaType(image)==IMA_BIT)
83  ? ( (IdImaDimX(image)+7)>>3 )
84  : (IdImaDimX(image)*IdSizeOfType(image)) ;
85  /* Calcul du resultat *****/
86  for ( y=0; y<IdImaDimY(image); y++ )
87  memset( image[y], 0, taille_ligne );
88 
89 IdImaSetUsedNbX(image,0);
90 IdImaSetUsedNbY(image,0);
91 
92  return image;
93 }
