125  antz = p3 * oldantx - p6 * oldanty + p9 * antz;
126  expr->magstate[0][ix][iy][iz] = antx;
127  expr->magstate[1][ix][iy][iz] = anty;
128  expr->magstate[2][ix][iy][iz] = antz;
129  }

131  expr->to = expr->to + tau;
132  if (alpha > M_PI/2) expr->to = expr->to * -1;
